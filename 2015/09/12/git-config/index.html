config

touch ~/.ssh/config

Host github.com
    User xxx
    IdentityFile ~/.ssh/id_rsa
    PreferredAuthentications publickey

Host git.coding.net
    HostName git.coding.net
    User xxx
    IdentityFile ~/.ssh/id_rsa

Host git.coding.net
    HostName git.coding.net
    User xxx2
    IdentityFile ~/.ssh/id_rsa.coding
